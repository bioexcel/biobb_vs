37 class BindingSite(BiobbObject):
38 """
39 | biobb_vs BindingSite
40 | This class finds the binding site of the input_pdb.
41 | Finds the binding site of the input_pdb_path file based on the ligands' location of similar structures (members of the sequence identity cluster)
42  
43 Args:
44 input_pdb_path (str): Path to the PDB structure where the binding site is to be found. File type: input. `Sample file <https://github.com/bioexcel/biobb_vs/raw/master/biobb_vs/test/data/utils/bindingsite.pdb>`_. Accepted formats: pdb (edam:format_1476).
45 input_clusters_zip (str): Path to the ZIP file with all the PDB members of the identity cluster. File type: input. `Sample file <https://github.com/bioexcel/biobb_vs/raw/master/biobb_vs/test/data/utils/bindingsite.zip>`_. Accepted formats: zip (edam:format_3987).
46 output_pdb_path (str): Path to the PDB containig the residues belonging to the binding site. File type: output. `Sample file <https://github.com/bioexcel/biobb_vs/raw/master/biobb_vs/test/reference/utils/ref_output_bindingsite.pdb>`_. Accepted formats: pdb (edam:format_1476).
47 properties (dic - Python dictionary object containing the tool parameters, not input/output files):
48 * **ligand** (*str*) - (None) Ligand to be found in the protein structure. If no ligand provided, the largest one will be selected, if more than one.
49 * **radius** (*float*) - (5.0) [0.1~1000|0.1] Cut-off distance (Ångstroms) around ligand atoms to consider a protein atom as a binding site atom.
50 * **max_num_ligands** (*int*) - (15) [0~1000|1] Total number of superimposed ligands to be extracted from the identity cluster. For populated clusters, the restriction avoids to superimpose redundant structures. If 0, all ligands extracted will be considered.
51 * **matrix_name** (*str*) - ("BLOSUM62") Substitution matrices for use in alignments. Values: BENNER22, BENNER6, BENNER74, BLASTN, BLASTP, BLOSUM45, BLOSUM50, BLOSUM62, BLOSUM80, BLOSUM90, DAYHOFF, FENG, GENETIC, GONNET1992, HOXD70, JOHNSON, JONES, LEVIN, MCLACHLAN, MDM78, MEGABLAST, NUC.4.4, PAM250, PAM30, PAM70, RAO, RISLER, SCHNEIDER, STR, TRANS.
52 * **gap_open** (*float*) - (-10.0) [-1000~1000|0.1] Gap open penalty.
53 * **gap_extend** (*float*) - (-0.5) [-1000~1000|0.1] Gap extend penalty.
54 * **remove_tmp** (*bool*) - (True) [WF property] Remove temporal files.
55 * **restart** (*bool*) - (False) [WF property] Do not execute if output files exist.
56 * **sandbox_path** (*str*) - ("./") [WF property] Parent path to the sandbox directory.
57  
58 Examples:
59 This is a use example of how to use the building block from Python::

153 return 0

157 fu.log(
158 "Loading input PDB structure %s" % (self.io_dict["in"]["input_pdb_path"]),

161 )

163 parser = Bio.PDB.PDBParser(QUIET=True)
164 structPDB = parser.get_structure(
165 structure_name, self.io_dict["in"]["input_pdb_path"]
166 )
167

169 structPDB = structPDB[0]
170  
171 # Use only one chain
172 n_chains = structPDB.get_list()
173 if len(n_chains) != 1:
174 fu.log(
175 "More than one chain found in the input PDB structure. Using only the first chain to find the binding site",
176 self.out_log,
177 self.global_log,
178 )
179 # get first chain in case there is more than one chain
180 for struct_chain in structPDB.get_chains():
181 structPDB = struct_chain

183 # Get AA sequence
184 structPDB_seq = get_pdb_sequence(structPDB)
185 if len(structPDB_seq) == 0:
186 fu.log(
187 self.__class__.__name__
  • W503 Line break before binary operator

190 self.out_log,

192 raise SystemExit(
193 self.__class__.__name__
  • W503 Line break before binary operator
194 + ": Cannot extract AA sequence from the input PDB structure %s. Wrong format?"
195 % self.io_dict["in"]["input_pdb_path"]
196 )
197 else:
198 fu.log(
199 "Found %s residues in %s"
200 % (len(structPDB_seq), self.io_dict["in"]["input_pdb_path"]),
201 self.out_log,
202 )
203  
204 # create temporary folder for decompressing the input_clusters_zip file
205 unique_dir = PurePath(fu.create_unique_dir())
206 fu.log(
207 "Creating %s temporary folder" % unique_dir, self.out_log, self.global_log
208 )
209  
210 # decompress the input_clusters_zip file
211 cluster_list = fu.unzip_list(
212 zip_file=self.io_dict["in"]["input_clusters_zip"],
213 dest_dir=str(unique_dir),
214 out_log=self.out_log,
215 )
216  
217 clusterPDB_ligands_aligned = []
218 clusterPDB_ligands_num = 0
219  
220 fu.log("Iterating on all clusters:", self.out_log)
221  
222 for idx, cluster_path in enumerate(cluster_list):
223 cluster_name = PurePath(cluster_path).stem
224 fu.log(" ", self.out_log)
225 fu.log(
226 "------------ Iteration #%s --------------" % (idx + 1), self.out_log
227 )
228 fu.log("Cluster member: %s" % cluster_name, self.out_log)
229  
230 # Load and Parse PDB
231 clusterPDB = {}
232 clusterPDB = parser.get_structure(cluster_name, cluster_path)[0]
233  
234 # Use only the first chain
235 for cluster_chain in clusterPDB.get_chains(): # type: ignore
236 clusterPDB = cluster_chain
237  
238 # Looking for ligands
239 clusterPDB_ligands = get_ligand_residues(clusterPDB)
240 if (len(clusterPDB_ligands)) == 0:
241 fu.log(
242 "No ligands found that could guide the binding site search. Ignoring this member: %s"
243 % cluster_name,
244 self.out_log,
245 )
246 continue
247  
248 # Selecting the largest ligand, if more than one
249 lig_atoms_num = 0
250 clusterPDB_ligand = {}
251 if self.ligand:
252 if self.ligand in [x.get_resname() for x in clusterPDB_ligands]:
253 for lig in clusterPDB_ligands:
254 if lig.get_resname() == self.ligand:
255 clusterPDB_ligand = lig
256 lig_atoms_num = len(lig.get_list())
257 fu.log(
258 "Ligand found: %s (%s atoms)"

260 self.out_log,
261 )
262 else:
263 fu.log(
264 "Ligand %s not found in %s cluster member, skipping this cluster"
265 % (self.ligand, cluster_name),
266 self.out_log,
267 )
268 continue
269 else:
270 if len(clusterPDB_ligands) > 1:
271 for lig_res in clusterPDB_ligands:
272 lig_res_atoms_num = len(lig_res.get_list())
273 fu.log(
274 "Ligand found: %s (%s atoms)"
275 % (lig_res.get_resname(), lig_res_atoms_num),
276 self.out_log,
277 )
278 if lig_res_atoms_num > lig_atoms_num:
279 clusterPDB_ligand = lig_res
280 lig_atoms_num = lig_res_atoms_num

282 clusterPDB_ligand = clusterPDB_ligands[0]
283 lig_atoms_num = len(clusterPDB_ligands[0].get_list())

285 fu.log(
286 "Member accepted. Valid ligand found: %s (%s atoms)"

289 )

297 aln, residue_map = align_sequences(

312 aln[0], aln[1]

315 fu.log("Gap less identity (%%): %s" % (gap_seq_identity), self.out_log)
316  
317 # Selecting aligned CA atoms from first model, first chain
318  
319 struct_atoms = []
320 cluster_atoms = []
321

323 try:
324 cluster_atoms.append(clusterPDB[residue_map[struct_res]]["CA"])

  • W503 Line break before binary operator

345 self.__class__.__name__
  • W503 Line break before binary operator

349 else:

354  
355 # Align against input structure
356  
357 si = Bio.PDB.Superimposer()
358 si.set_atoms(struct_atoms, cluster_atoms)

366 clusterPDB_ligands_aligned.append(clusterPDB_ligand_aligned)

368 # Stop after n accepted cluster members
369  
370 clusterPDB_ligands_num += 1

373 break

375 fu.log(" ", self.out_log)
376 fu.log("----------------------------------------", self.out_log)
377 fu.log(
378 "All transformed ligand coordinates saved, getting binding site residues",
379 self.out_log,
380 )
381  
382 # Select binding site atoms as those around cluster superimposed ligands
383  
384 fu.log(
385 "Defining binding site residues as those %sÅ around the %s cluster superimposed ligands"
386 % (self.radius, clusterPDB_ligands_num),
387 self.out_log,
388 )
389  
390 # select Atoms from aligned ligands
391 clusterPDB_ligands_aligned2 = [res for res in clusterPDB_ligands_aligned]
392 clusterPDB_ligands_aligned_atoms = Bio.PDB.Selection.unfold_entities(
393 clusterPDB_ligands_aligned2, "A"
394 )
395  
396 # select Atoms from input PDB structure
397 structPDB_atoms = [atom for atom in structPDB.get_atoms()]
398  
399 # compute neighbors for aligned ligands in the input PDB structure
400 structPDB_bs_residues_raw = {}
401 structPDB_neighbors = Bio.PDB.NeighborSearch(structPDB_atoms)
402 for ligand_atom in clusterPDB_ligands_aligned_atoms:
403 # look for PDB atoms 5A around each ligand atom
404 k_l = structPDB_neighbors.search(ligand_atom.coord, self.radius, "R")
405 for k in k_l:
406 structPDB_bs_residues_raw[k.get_id()] = k.get_full_id()
407  
408 # Save binding site to PDB
409  
410 io = Bio.PDB.PDBIO()
411 fu.log(
412 "Writing binding site residues into %s"
413 % (self.io_dict["out"]["output_pdb_path"]),
414 self.out_log,
415 )
416  
417 # unselect input PDB atoms not in binding site
418 structPDB_bs_atoms = 0
419 p = re.compile("H_|W_|W")
420 residue_ids_to_remove = []
421 for res in structPDB.get_residues():
422 if res.id not in structPDB_bs_residues_raw.keys():
423 # add residue to residue_ids_to_remove list
424 residue_ids_to_remove.append(res.id)
425 elif p.match(res.resname):
426 # add residue to residue_ids_to_remove list
427 residue_ids_to_remove.append(res.id)
428 else:
429 # this residue will be preserved
430 structPDB_bs_atoms += len(res.get_list())
431  
432 # unselect input PDB atoms not in binding site
433 for chain in structPDB:
434 for idr in residue_ids_to_remove:
435 chain.detach_child(idr)
436  
437 # write PDB file
438 io.set_structure(structPDB)
439 io.save(self.io_dict["out"]["output_pdb_path"])
440  
441 # Copy files to host
442 self.copy_to_host()
443  
444 self.tmp_files.extend(
445 [self.stage_io_dict.get("unique_dir", ""), str(unique_dir)]
446 )
447 self.remove_tmp_files()
448  
449 self.check_arguments(output_files_created=True, raise_exception=False)
450  
451 return 0
