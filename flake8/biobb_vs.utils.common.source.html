117 def align_sequences(
118 seqA, seqB, matrix_name="BLOSUM62", gap_open=-10.0, gap_extend=-0.5
119 ):
120 """
121 Performs a global pairwise alignment between two sequences using the Needleman-Wunsch algorithm as implemented in Biopython.
122 Returns the alignment and the residue mapping between both original sequences.
123 """
124  
125 # seq list to seq string
126 sequence_A = "".join([i[1] for i in seqA])
127 sequence_B = "".join([i[1] for i in seqB])
128  
129 # get matrix from matrix_name
130 # try:
131 # matrix = getattr(Bio.SubsMat.MatrixInfo, matrix_name)
132 # except AttributeError:
133  
134 matrix = Bio.Align.substitution_matrices.load(matrix_name)
135  
136 # print(Bio.SubsMat.MatrixInfo)
137 # print(type(substitution_matrices.select()))
138 # matrix = getattr(substitution_matrices.load(), matrix_name)
139  
140 # Do pairwaise alignment
141 alns = Bio.pairwise2.align.globalds(
142 sequence_A,
143 sequence_B,
144 matrix,
145 gap_open,
146 gap_extend,
147 penalize_end_gaps=(False, False),
148 )
149  
150 best_aln = alns[0]
151 aligned_A, aligned_B, score, begin, end = best_aln
152  
153 # Equivalent residue numbering. Relative to reference
154 mapping = {}
155 aa_i_A, aa_i_B = 0, 0
156 for aln_i, (aa_aln_A, aa_aln_B) in enumerate(zip(aligned_A, aligned_B)):
157 if aa_aln_A == "-":
158 if aa_aln_B != "-":
159 aa_i_B += 1
160 elif aa_aln_B == "-":
161 if aa_aln_A != "-":
162 aa_i_A += 1
163 else:
164 assert seqA[aa_i_A][1] == aa_aln_A
165 assert seqB[aa_i_B][1] == aa_aln_B
166 mapping[seqA[aa_i_A][0]] = seqB[aa_i_B][0]
167 aa_i_A += 1
168 aa_i_B += 1
169  
170 return ((aligned_A, aligned_B), mapping)

173 def calculate_alignment_identity(alignedA, alignedB):
174 """
175 Returns the percentage of identical characters between two sequences
176 """
177 matches = [alignedA[i] == alignedB[i] for i in range(len(alignedA))]
178 seq_id = (100 * sum(matches)) / len(alignedA)
179  
180 gapless_sl = sum(
181 [1 for i in range(len(alignedA)) if (alignedA[i] != "-" and alignedB[i] != "-")]
182 )
183 gap_id = (100 * sum(matches)) / gapless_sl
184 return (seq_id, gap_id)

187 def get_ligand_residues(
188 PDBchain,
189 ignore_wats=True,
190 ignore_small_molec=True,
191 ignore_ions=True,
192 ignore_modres=True,
193 ):
194 """
195 Returns heteroatoms residues.
196 Args:
197 PDBchain (Bio.PDB.PDBParser chain object): PDB selection of the chain to be scanned
198 ignore_wats (boolean): If True, water residues will be skipped, and not returned as ligand residues
199 ignore_small_molec (boolean): If True, small ligands (< 5 atoms) will be skipped, and not returned as ligand residues
200 ignore_ions (boolean): If True, ion residues will be skipped, and not returned as ligand residues
201 ignore_modres (boolean): If True, modified aminoa acid residues will be skipped, and not returned as ligand residues
202 """
203  
204 # small_molec_atoms_min = 5
205 ligands = []
206  
207 for res in PDBchain.get_residues():
208 res_entity = res.get_full_id()
209 res_hetflag = res_entity[3][0]
210  
211 # skip aminoacids
212 if res_hetflag == " ":
213 continue
214  
215 # skip waters, if defined
216 if res_hetflag == "W":
217 if not ignore_wats:
218 ligands.append(res)
219 continue
220 # skip small_molec (< small_molec_atoms_min)
221 # if ignore_small_molec:
222 # if len(res.get_list()) < small_molec_atoms_min:
223 # continue
224 # skip ions
225 if ignore_ions:
226 if res.get_resname().strip() in __ions():
227 continue
228 # skip modres
229 if ignore_modres:
230 if res.get_resname().strip() in __modres().keys():
231 continue
232  
233 # add as ligand
234 ligands.append(res)
235  
236 return ligands

988 def _from_string_to_list(input_data: Optional[Union[str, list[str]]]) -> list[str]:
989 """
990 Converts a string to a list, splitting by commas or spaces. If the input is already a list, returns it as is.
991 Returns an empty list if input_data is None.
992  
993 Parameters:
994 input_data (str, list, or None): The string, list, or None value to convert.
995  
996 Returns:
997 list: A list of string elements or an empty list if input_data is None.
998 """
999 if input_data is None:
1000 return []
1001  
1002 if isinstance(input_data, list):
1003 # If input is already a list, return it
1004 return input_data
1005  
1006 # If input is a string, determine the delimiter based on presence of commas
1007 delimiter = "," if "," in input_data else " "
1008 items = input_data.split(delimiter)
1009  
1010 # Remove whitespace from each item and ignore empty strings
1011 processed_items = [item.strip() for item in items if item.strip()]
1012  
1013 return processed_items
